CREATE TABLE category (
	id VARCHAR2(32 CHAR) PRIMARY KEY,
	name VARCHAR2(20 CHAR) UNIQUE NOT NULL,
	description VARCHAR2(100 CHAR) NOT NULL
);

CREATE TABLE pic (
	id VARCHAR2(32 CHAR) PRIMARY KEY,
	oldName VARCHAR2(50 CHAR) NOT NULL
);

CREATE TABLE book (
	id VARCHAR2(32 CHAR) PRIMARY KEY,
	name VARCHAR2(20 CHAR) NOT NULL,
	author VARCHAR2(50 CHAR) NOT NULL,
	description VARCHAR2(100 CHAR) NOT NULL,
	picId VARCHAR2(32 CHAR) NOT NULL,
	categoryId VARCHAR2(32 CHAR) NOT NULL
);

ALTER TABLE book ADD CONSTRAINTS fk_book_picId FOREIGN KEY (picId) REFERENCES pic(id);
ALTER TABLE book ADD CONSTRAINTS fk_book_categoryId FOREIGN KEY (categoryId) REFERENCES category(id);
