CREATE TABLE category (
	id VARCHAR2(32 CHAR) PRIMARY KEY,
	name VARCHAR2(20 CHAR) UNIQUE NOT NULL,
	description VARCHAR2(100 CHAR) NOT NULL
);

CREATE TABLE book (
	id VARCHAR2(32 CHAR) PRIMARY KEY,
	name VARCHAR2(100 CHAR) NOT NULL,
	author VARCHAR2(100 CHAR) NOT NULL,
	price NUMBER(12, 2) NOT NULL,
	description VARCHAR2(100 CHAR) NOT NULL,
	pic VARCHAR2(100 CHAR) NOT NULL,
	categoryId VARCHAR2(32 CHAR)
);

ALTER TABLE book ADD CONSTRAINTS fk_book_categoryId FOREIGN KEY (categoryId) REFERENCES category(id);

CREATE TABLE iaccount (
	id VARCHAR2(32) PRIMARY KEY,
	code VARCHAR2(32) NOT NULL,
	name VARCHAR2(100 CHAR) UNIQUE NOT NULL,
	password VARCHAR2(100 CHAR) NOT NULL,
	cellphone VARCHAR2(11 CHAR) NOT NULL,
	email VARCHAR2(50 CHAR) NOT NULL,
	address VARCHAR2 (100 CHAR) NOT NULL,
	status NUMBER(2) NOT NULL
);
